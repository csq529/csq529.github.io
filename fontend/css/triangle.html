    # 1、transform: rotate 配合 overflow: hidden 绘制三角形

    # 完整代码

    .triangle {
    width: 22px;
    height: 13px;
    position: relative;
    overflow: hidden;
    
    &::before {
        content: "";
        position: absolute;
        top: 0;
        left: 0;
        right: 0;
        bottom: 0;
        background: red;
        transform-origin: left bottom;
        transform: rotate(45deg);
    }
}

    # 2、使用border绘制三角形

    # 其中一个边框的颜色是想要设置的颜色，其他3边为transparent. top ===> 上箭头 left ===> 右箭头

    .triangle2 {
  width: 0;
  height: 0;
  border-top: 10px solid red;
  border-bottom: 10px solid transparent;
  border-left: 10px solid transparent;
  border-right: 10px solid transparent;
}

    # 3.使用线性渐变linear-gradient绘制三角形

      <div class="normal"></div>
  <div class="gradient"></div>
  <div class="rotate"></div>
  <div class="top"></div>
  <div class="left"></div>
  <div class="right"></div>
  <div class="bottom"></div>

     .normal {
    background: linear-gradient(45deg, red, yellow)
  }

  .gradient {
    background: linear-gradient(45deg, red, red 50%, yellow 50%, yellow 100%)
  }

  .rotate {
    background: linear-gradient(45deg, red, red 50%, transparent 50%, transparent 100%)
  }

  .top {
    background: linear-gradient(45deg, red, red 50%, transparent 50%, transparent 100%);
    transform: rotate(135deg)
  }

  .left {
    background: linear-gradient(45deg, red, red 50%, transparent 50%, transparent 100%)
    transform: rotate(225deg)
  }

  .right {
    background: linear-gradient(45deg, red, red 50%, transparent 50%, transparent 100%)
    transform: rotate(45deg)
  }

  .bottom {
    background: linear-gradient(45deg, red, red 50%, transparent 50%, transparent 100%)
    transform: rotate(-45deg)
  }
