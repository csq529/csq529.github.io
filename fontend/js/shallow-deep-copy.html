# 前言

在js中，数据类型分为基本数据类型和引用数据类型两种，对于基本数据类型来说，它的值直接存储在栈内存中，而对于引用数据类型来说，它在栈内存中仅仅存储了一个引用，而真正的数据存储在堆内存中。

# 基本数据类型

var a = 1
var b = a
b = 2
console.log(a) // 1
console.log(b) // 2

可以看到对于基本数据类型来说，将一个基本类型的值赋值给变量a，接着将变量a的值赋值给变量b，然后修改b，可以看到b的只修改了，而a的值没有被修改，两个变量都是用的是独立的数据。

# 引用数据类型

var obj = {
  a: 1,
  b: 2
}
var obj1 = obj
obj1.a = 2
console.log(obj.a) // 2
console.log(obj1.a) // 2

可以看到两个对象的值都被改了。

对象是引用数据类型，对于引用数据类型来说，将obj赋值给obj1的时候，其实仅仅只是将obj存储在栈内存中的引用赋予了obj1，而两个对象此时指向的是堆内存中的同一个数据，所以当修改任意一个值的时候，修改的都是堆内存中的数据，而不是引用。所以只要修改了，同样引用的对象的值也发生了改变。

# 浅拷贝

其实上面的那个例子就是一个浅拷贝，对于浅拷贝而言，就是只拷贝对象的引用(内存地址)，而不是深层次的拷贝对象的值，多个对象指向堆内存中的同一个对象，任何一个修改都会使得所有对象的值修改，因为他们公用一条数据。

# 深拷贝

深拷贝是将一个对象从内存中完整的拷贝一份出来，从堆内存中开辟一个新的区域存放新对象，且修改新对象不会影响元对象。

总而言之，浅拷贝只复制指向某个对象的指针，而不复制对象本身，新旧对象还是共享同一块内存；但深拷贝会创造一个一摸一样的对象，新对象和原对象不共享内存，修改新对象不会改到原对象。

# 赋值和深/浅拷贝的区别

这三者的区别如下，不过比较的前提都是针对引用类型：

  • 赋值：当把一个对象赋值给一个新的变量时，赋的其实就是该对象在栈中的地址，而不是堆中的数据。也就是两个对象指向同一个储存空间，无论哪个对象发生改变，其实都是改变的储存空间的内容，因此，两个对象是联动的。

  • 浅拷贝：重新在堆中创建内存，拷贝前后对象的基本数据类型互不影响，当拷贝前后对象的引用类型，因共享同一块内存，会相互影响。

  • 深拷贝：从堆内存中开辟一个新的区域存放新对象，对对象中的子对象进行递归拷贝，拷贝前后的两个对象互不影响。

根据以下的例子，来对比一下赋值与深/浅拷贝得到的对象修改后对原始对象的影响。

// 对象赋值
let obj1 = {
  a: 1,
  arr: [1, [2]]
}
let obj2 = obj1
obj2.a = 2
obj.arr[1] = [3]

console.log('obj1:', obj1) // obj1: { a: 2, arr: [1, [3]] }
console.log('obj2:', obj2) // obj2: { a: 2, arr: [1, [3]] }

// 浅拷贝
let obj1 = {
  a: 1,
  arr: [1, [2]]
}
let obj3 = shallowClone(obj1)
obj3.a = 2
obj3.arr[1] = 3

function shallowClone(obj) {
  var target = {}
  for(var i in obj) {
    if(obj.hasOwnProperty(i)) {
      target[i] = obj[i]
    }
  }
  return target
}

console.log('obj1:', obj1) // obj1: { a: 1, arr: [1, [3]] }
console.log('obj3:', obj3) // obj3: { a: 2, arr: [1, [3]] }

// 深拷贝
let obj1 = {
  a: 1,
  arr: [1, [2]]
}
let obj4 = deepClone(obj1)
obj4.a = 2
obj4.arr[1] = 3

function deepClone(obj) {
  if (obj === 'null') return obj
  if (obj instanceof Date) return new Date(obj)
  if (obj instanceof RegExp) return new RegExp(obj)
  if (typeof obj !== 'object') return obj
  let cloneObj = new obj.constructor()
  for (let key in obj) {
    if (obj.hasOwnProperty(key)) {
      cloneObj[key] = deepClone(obj[key])
    }
  }
  return cloneObj
}

console.log('obj1:', obj1) // obj1: { a: 1, arr: [1, [2]] }
console.log('obj4:', obj4) // obj4: { a: 2, arr: [1, [3]] }

# 浅拷贝的实现方式

1.Object.assign() Object.assign()方法用于将所有可枚举属性的值从一个或多个源对象复制到目标对象。它将返回目标对象。

let obj1 = { a: [1], b: 2 }
let obj2 = Object.assign({}, obj1)
obj2.a[0] = 2
obj2.b = 3
console.log('obj1:', obj1) // { a: [2], b: 2 }
console.log('obj2:', obj2) // { a: [2], b: 3 }

2.函数库lodash的_.clone()方法 3.展开运算符... 展开运算符是一个ES6特性，它提供了一种非常方便的方式来执行浅拷贝，这与Object.assign()的功能相同。

let obj1 = {
  name: 'csq',
  address: {
    x: 200,
    y: 200
  }
}
let obj2 = { ...obj1 }
obj2.name = 'gjl'
obj2.address.x = 300
console.log('obj1:', obj1) // { name: 'csq', address: { x: 300, y: 200 } }
console.log('obj2:', obj2) // { name: 'gjl', address: { x: 300, y: 200 } }

4.Array.prototype.concat()

let arr1 = [
  1,
  2,
  {
    name: 'csq'
  }
]
let arr2 = arr1.concat()
arr2[2].name = 'gjl'
arr2[0] = 2
console.log(arr1) // [1, 2, { name: 'gjl' }]
console.log(arr2) // [2, 2, { name: 'gjl' }]

5.Array.prototype.slice()

let arr1 = [
  1,
  2,
  {
    name: 'csq'
  }
]
let arr2 = arr1.slice()
arr2[0] = 2
arr2[2].name = 'gjl'
console.log(arr1) // [1, 2, { name: 'gjl' }]
console.log(arr2) // [2, 2, { name: 'gjl' }]

# 深拷贝的实现方式

1.JSON.parse(JSON.stringify())

let arr1 = [
  1,
  2,
  {
    name: 'csq'
  }
]
let arr2 = JSON.parse(JSON.stringify(arr1))
arr2[0] = 2
arr2[2].name = 'gjl'
console.log(arr1) // [1, 2, { name: 'csq' }]
console.log(arr2) // [2, 2, { name: 'gjl' }]

这也是利用JSON.stringfy()将对象转成JSON字符串，再用JSON.parse()把字符串解析成对象，一去一来，新的对象就产生了，而且对象会开辟新的栈，实现深拷贝。

这种方法虽然可以实现数组或对象深拷贝，但不能处理函数和正则,因为这两者基于JSON.stringify和JSON.parse处理之后，得到的正则就不是正则（变为空对象），得到的函数就不是函数了（变为null）了。

function deepClone(obj) {
  // if (obj === 'null') return obj
  // if (obj instanceof Date) return new Date(obj)
  // if (obj instanceof RegExp) return new RegExp(obj)
  // if (typeof obj !== 'Object') return obj
  let cloneObj = isArray(obj) ? [] : {}
  for(let key in obj) {
    if (isObject(obj[key])) {
      cloneObj[key] = deepClone(obj[key])
    } else {
      cloneObj[key] = obj[key]
    }
  }
  return cloneObj
}

function isArray(val) {
  return Object.prototype.toString.call(val) === '[Object Array]'
}

function isObject(val) {
  return typeof val === 'object' && val !== 'null'
}

let obj1 = {
  x: 1,
  y: 2,
  z: {
    a: 3,
    b: 4
  }
}
let obj2 = deepClone(obj1)
obj2.x = 11
obj2.z.a = 4
console.log(obj1) // { x: 1, y: 2, z: { a: 3, b: 4 } }
console.log(obj2) // { x: 11, y: 2, z: { a: 4, b: 4 } }
