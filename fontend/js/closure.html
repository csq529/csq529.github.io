# 闭包概念

闭包就是能够读取其他函数内部变量的函数。本质上，闭包就是将函数内部和函数外部连接起来的桥梁。

# 闭包的作用

  • 保存变量于内存中，避免全局变量的污染
function outFn() {
  var count = 0
  function insideFn() {
    return ++count
  }
  return insideFn
}
var fn = outFn()
fn() // 1
fn() // 2
fn() // 3
  • 在函数外部读取函数内部的变量
var name = 'csq'
function fn1() {
  var age = 18
  function fn2() {
    var sex = '男'
    console.log(name, age, sex)
  }
  return fn2
}
var fn = fn1()
fn() // csq 18 男
  • 对外提供共有属性和方法
function parentFun() {
  var name = 'csq'
      age = 18
      obj = {
        name,
        age,
        allInfo() {
          return {
            name,
            age
          }
        }
      }
  return obj.allInfo
}
var fn = parentFun()
fn() // { name: 'csq', age: 18 }

# 闭包的应用场景

循环中找到对应的元素的索引js代码

/*
*原本是想点击哪个按钮，正确的显示对应哪个按钮元素的索引值，但是无论点击哪个按钮都是最后一个索引。
*错误原因是：因为for循环执行的速度很快，执行结束后，变量i的值已经变成了最后一项，所以当点击事件执行时，读取的信息总是最后一项
*/
window.onload = function() {
  var btns = document.getElementsByTagName('input')
  for (var i = 0; i < btns.length; i++) {
    btns[i].onclick = function() {
      console.log(i) // 3 3 3
    }
  }
}

// html代码
<input type="button" value="按钮1" name=""/>
<input type="button" value="按钮2" name=""/>
<input type="button" value="按钮3" name=""/>

解决方法1：使用匿名函数自执行

windiw.onload = function() {
  var btns = document.getElementsByTagName('input')
  for (var i = 0; i < btns.length; i++) {
    (function(i) {
      btns[i].onclick = function() {
        console.log(i) // 0 1 2
      }
    })(i)
  }
}

解决方法2：使用let

window.onload = function() {
  var btns = document.getElementsByTagName('input')
  for(let i = 0; i < btns.length; i++) {
    btns[i].onclick = function() {
      console.log(i) // 0 1 2 
    }
  }
}
